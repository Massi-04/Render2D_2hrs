#version 330 core

layout(location = 0) out vec4 color;

in vec3 v_Color;
in vec2 v_TexCoord;
in float v_TexIndex;
uniform sampler2D u_TexSlots[32];

void main()
{
    if(v_TexIndex < 0.0f)
    {
        color = vec4(v_Color, 1.0f);
    }
    else
    {
        color = texture(u_TexSlots[int(v_TexIndex + 0.5)], v_TexCoord) * vec4(v_Color, 1.0f);
    }
}
